#!/usr/bin/env python3
import requests
import json
import os

def test_download_files():
    """测试不同大小文件的下载"""
    
    # 配置
    BASE_URL = "http://localhost:5000"  # 根据实际端口调整
    TEST_FILES = [
        {"name": "kiosk.mnu", "expected_size": "3.61KB", "type": "small"},
        {"name": "network.zip", "expected_size": "5.44GB", "type": "large"},
        {"name": "EVE.zip", "expected_size": "28.91GB", "type": "huge"}
    ]
    
    # 登录获取会话
    session = requests.Session()
    
    # 登录
    login_data = {
        "username": "xuzean",  # 替换为实际用户名
        "password": "your_password",  # 替换为实际密码
        "captcha": "1234"  # 需要先获取验证码
    }
    
    # 先获取验证码
    captcha_resp = session.get(f"{BASE_URL}/get_captcha")
    if captcha_resp.status_code == 200:
        captcha_data = captcha_resp.json()
        login_data["captcha"] = captcha_data.get("captcha", "1234")
    
    # 登录
    login_resp = session.post(f"{BASE_URL}/login", json=login_data)
    if login_resp.status_code != 200:
        print("❌ 登录失败")
        return
    
    print("✅ 登录成功")
    
    # 测试文件下载
    for test_file in TEST_FILES:
        filename = test_file["name"]
        print(f"\n🔍 测试 {test_file['type']} 文件: {filename}")
        
        # 检查文件信息
        info_resp = session.get(f"{BASE_URL}/file_info/{filename}")
        if info_resp.status_code == 200:
            file_info = info_resp.json()
            print(f"✅ 文件信息: {file_info}")
        else:
            print(f"❌ 无法获取文件信息: {info_resp.status_code}")
            continue
        
        # 尝试下载
        download_url = f"{BASE_URL}/download/{filename}"
        print(f"📥 下载URL: {download_url}")
        
        try:
            # 使用流式下载，只获取前1MB来测试
            with session.get(download_url, stream=True) as resp:
                if resp.status_code == 200:
                    # 读取前1MB来验证下载正常
                    content = b""
                    for chunk in resp.iter_content(chunk_size=8192):
                        content += chunk
                        if len(content) >= 1024 * 1024:  # 1MB
                            break
                    
                    actual_size = len(content)
                    print(f"✅ {filename} 下载测试成功，接收数据: {actual_size} bytes")
                    
                    # 保存测试文件片段
                    test_dir = "/tmp/download_test"
                    os.makedirs(test_dir, exist_ok=True)
                    test_file_path = os.path.join(test_dir, f"test_{filename}")
                    
                    with open(test_file_path, 'wb') as f:
                        f.write(content)
                    
                    print(f"💾 测试数据保存到: {test_file_path}")
                    
                else:
                    print(f"❌ {filename} 下载失败: {resp.status_code}")
                    print(f"错误信息: {resp.text}")
                    
        except Exception as e:
            print(f"❌ {filename} 下载异常: {str(e)}")

if __name__ == '__main__':
    test_download_files()
